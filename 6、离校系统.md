	public String clobToString(Clob clob) throws SQLException, IOException { 
        String reString = ""; 
        if(clob!=null) {
        	java.io.Reader is = clob.getCharacterStream();// 得到流 
            BufferedReader br = new BufferedReader(is); 
            String s = br.readLine(); 
            StringBuffer sb = new StringBuffer(); 
            while (s != null) {// 执行循环将字符串全部取出付值给StringBuffer由StringBuffer转成STRING 
                sb.append(s); 
                s = br.readLine(); 
            } 
            reString = sb.toString(); 
        }
        return reString; 
    }

	@Route
	public boolean saveOrUpdate( ESurestream surestream ){

		String accountName=surestream.getString("ACCOUNT_NAME");
		
		ValidatorSupport.isEmpty(accountName, "【账号名称】不能为空");
		try{
			BaseAccount bAccount=null;
			if(!Validate.isEmpty(accountUid)){
				bAccount=this.get(BaseAccount.class, accountUid);
			}
			
			if(BaseValidateUtils.accountIsUseing(this.getSession(), accountName, accountUid)) {
				throw new RWarn(RWarn._999897, "【账号名称】已存在！");
			}
			
			Date nowDate=DateSource.getTradedate();
			if(!Validate.isEmpty(bAccount)&&!Validate.isEmpty(bAccount.getAccountUid())){
				String oldPw=bAccount.getAccountPw();
				this.hashMap2Obj(surestream.getDownbuf(), bAccount);
		//		DataDetailUtils.map2obj(surestream.getDownbuf(), bAccount);
				if(!Validate.isEmpty(accountPw)){
					accountPw = MD5.md5Password(accountPw);
				}else{
					accountPw=oldPw;
				}
				bAccount.setAccountPw(accountPw);
				if(Validate.isEmpty(bAccount.getAccountInfoUid())) {
					bAccount.setAccountInfoUid(bAccount.getAccountUid());
				}
				bAccount.setUpdateTime(nowDate);
				bAccount.setUpdateUid(Accountmgt.getAccountUid(surestream));
				this.update(bAccount);
			}else{
				bAccount=new BaseAccount();
				this.hashMap2Obj(surestream.getDownbuf(), bAccount);
			//	DataDetailUtils.map2obj(surestream.getDownbuf(), bAccount);
				if(!Validate.isEmpty(accountPw)){
					accountPw = MD5.md5Password(accountPw);
				}else{
					//设置默认密码
					String defaultPW=DicConst.DEFAULT_PASSWORD;;
					accountPw = MD5.md5Password(defaultPW);//默认密码
				}
				bAccount.setAccountUid(DcKeyGenerate.getStringKeyCode("accountid"));
				bAccount.setAccountInfoUid(bAccount.getAccountUid());
				bAccount.setDataStatus("1");
				bAccount.setAccountPw(accountPw);

				bAccount.setErrorCount(0);
				bAccount.setUpdateTime(nowDate);
				bAccount.setUpdateUid(Accountmgt.getAccountUid(surestream));
				bAccount.setCreateTime(nowDate);
				bAccount.setCreateUid(Accountmgt.getAccountUid(surestream));
				this.save(bAccount);
				accountUid=bAccount.getAccountUid();
			}
			
			//账户信息
			BaseAccountInfo accountInfo=null;
			if(!Validate.isEmpty(bAccount.getAccountInfoUid())) {
				accountInfo=this.get(BaseAccountInfo.class, bAccount.getAccountInfoUid());
			}
			if(!Validate.isEmpty(accountInfo)&&!Validate.isEmpty(accountInfo.getInfoUid())) {
				accountInfo.setCellPhone(surestream.getString("CELL_PHONE"));
				accountInfo.setCollegeUid(surestream.getString("COLLEGE_UID"));
				accountInfo.setEmail(surestream.getString("EMAIL"));
				accountInfo.setSex(surestream.getString("SEX"));
				accountInfo.setRealName(bAccount.getAccountNickname());
				accountInfo.setUpdateTime(nowDate);
				accountInfo.setUpdateUid(accountUid);
				this.update(accountInfo);
			}else {
				accountInfo=new BaseAccountInfo();
				accountInfo.setInfoUid(bAccount.getAccountUid());
				accountInfo.setCellPhone(surestream.getString("CELL_PHONE"));
				accountInfo.setCollegeUid(surestream.getString("COLLEGE_UID"));
				accountInfo.setEmail(surestream.getString("EMAIL"));
				accountInfo.setRealName(bAccount.getAccountNickname());
				accountInfo.setRegTime(bAccount.getCreateTime());
				accountInfo.setSex(surestream.getString("SEX"));
				accountInfo.setUpdateTime(nowDate);
				accountInfo.setUpdateUid(accountUid);
				this.save(accountInfo);
			}
		}catch ( RWarn e) {
	        TryThrowable.agent( e, this );
	    }catch ( RException e) {
	        TryThrowable.agent( e, this );
	    }catch (Exception e) {
	        e.printStackTrace();
	    	TryThrowable.agent( e, this );
	    }
	    surestream.put("accountUid", accountUid);
		surestream.putReason(RWarn._000000,"成功");
		return true;
	}

	@Route
	public boolean updatePW(ESurestream surestream) {
		String oldPassword=surestream.getString("oldPassword");
		String password=surestream.getString("password");
		String repassword=surestream.getString("repassword");
		if (Validate.isEmpty(oldPassword)) {
			throw new RWarn( RWarn._999897, "【原始密码】不能为空" );
		}
		if (Validate.isEmpty(password)) {
			throw new RWarn( RWarn._999897, "【新密码】不能为空" );
		}
		if (Validate.isEmpty(repassword)) {
			throw new RWarn( RWarn._999897, "【重复新密码】不能为空" );
		}
		if(!password.equals(repassword)){
			throw new RWarn( RWarn._999897, "【两次输入密码】不一致，请重新输入" );
		}
		
		String accountUid=Accountmgt.getAccountUid(surestream);
		if(!Validate.isEmpty(accountUid)){
			BaseAccount adminAccount=this.get(BaseAccount.class, accountUid);
			if(!Validate.isEmpty(adminAccount)&&!Validate.isEmpty(adminAccount.getAccountUid())){
				//修改密码
				if (CiSSO.c_isso_switch()) {
					com.easy.isso.client.util.HashMap downbuf = MSiSSO.request( surestream.request(), "992001"
							, new EParam("srcpwd", oldPassword)
							, new EParam("newpwd", password)
							);
					if(RWarn._000000.equals(downbuf.getString("reasoncode"))) {
						adminAccount.setAccountPw(MD5.md5Password(password));
						adminAccount.setUpdateTime(DateSource.getTradedate());
						adminAccount.setUpdateUid(Accountmgt.getAccountUid(surestream));
						this.update(adminAccount);
					}else {
						throw new RWarn(downbuf.getString("reasoncode"), downbuf.getString("reasonremark"));
					}
				}else {
					oldPassword = MD5.md5Password(oldPassword);
					if(!oldPassword.equals(adminAccount.getAccountPw())){
						throw new RWarn( RWarn._999897, "【原始密码】输入不正确" );
					}
					password = MD5.md5Password(password);
					adminAccount.setAccountPw(password);
					adminAccount.setUpdateTime(DateSource.getTradedate());
					adminAccount.setUpdateUid(Accountmgt.getAccountUid(surestream));
					this.update(adminAccount);
				}
			}
		}
		surestream.putReason(RWarn._000000,"密码修改操作成功");
        return true;
	}

	public boolean getCodeMap( ESurestream surestream ) {

		String roleType = surestream.getString("rolePowerType");
		ValidatorSupport.isEmpty(accountName, "【账号】不能为空");
		ValidatorSupport.isEmpty(roleType, "【数据权限代码】不能为空");
		
		HashMap accountMap = this.queryForMap("SELECT * FROM BASE_ACCOUNT WHERE ACCOUNT_NAME =? AND DATA_STATUS=? ", accountName,"1");
		
		if(accountMap==null) {
			throw new RWarn(RWarn._999897, "【账号信息】不存在");
		}
		accountMap.remove("ACCOUNT_PW");
		
		if(RoleTypeEnum.college.getKey().equals(roleType)) {
			String sql = " SELECT T.CODE_UID FROM BASE_ACCOUNT_POWER T WHERE T.ACCOUNT_UID=? AND T.CODE_TYPE=? ";
			List<HashMap> powerList = this.sqlQuery(sql, accountMap.getString("ACCOUNT_UID"),roleType);
			String powerCodes="";
			if(powerList!=null&&powerList.size()>0){
				StringBuilder powerStr = new StringBuilder("");
				for(int i=0;i<powerList.size();i++){
					HashMap temp=powerList.get(i);
					powerStr.append(",").append(temp.getString("CODE_UID"));
				}
				powerCodes=powerStr.substring(1);
			}
			accountMap.put("powerStr", powerCodes);
		}else if(RoleTypeEnum.counselor.getKey().equals(roleType)) {
			String sql = " SELECT T.CODE_UID,C.CLASS_NAME FROM BASE_ACCOUNT_POWER T LEFT JOIN BASE_SYS_CLASS C ON C.CLASS_CODE=T.CODE_UID WHERE T.ACCOUNT_UID=? AND T.CODE_TYPE=? ";
			List<HashMap> powerList = this.sqlQuery(sql, accountMap.getString("ACCOUNT_UID"),roleType);
			List<HashMap> initList = new ArrayList<HashMap>();
			if(powerList!=null&&powerList.size()>0){
				HashMap temp = null;
				for(HashMap poMap:powerList){
					temp = new HashMap();
					temp.put("title", poMap.getString("CLASS_NAME"));
					temp.put("value", poMap.getString("CODE_UID"));
					initList.add(temp);
				}
			}
			
			accountMap.put("initList", initList.toArray());
		}
		accountMap.put("INPUT_POWER_TYPE", roleType);
		surestream.put("data", accountMap);
		surestream.putReason(RWarn._000000,"查询成功");
		return true;
	}

	            ExcelHead head = new ExcelHead();
	            head.setRowCount(1); // 头所占行数
	            head.setColumns(excelColumns);  // 列的定义
	            head.setColumnsConvertMap(excelColumnsConvertMap); // 列的转换

	            List<Map<String, Object>> classList=null;
	    		try {
	    			classList = ExcelHelper.getInstanse().importToObjectList(head, multipartFile.getInputStream());
	    		} catch (Exception e) {
	    			throw new RWarn( RWarn._999897,"导入班级数据失败" ); 
	    		}
	    		
	            String specialtyCode = null;//专业代码
	            String collegeCode = null;//学院代码
    	        String remark = null;
	            int succnt = 0 ;
	            int failcnt = 0;
	    		
	    		for(Map<String, Object>  stuObj : classList){
	    			collegeCode = null;
	    			specialtyCode = null;
	    			remark = "";
	    	       	int errcnt = 0;
	        	    if (Validate.isEmpty(stuObj.get("CLASS_NO"))){
	        	    	remark += "班号不能为空 .";
	        	    	errcnt++;
	        	    }
	        	    if (Validate.isEmpty(stuObj.get("CLASS_NAME"))){
	        	    	remark += "班级名称不能为空 .";
	        	    	errcnt++;
	        	    }
	        	    if (Validate.isEmpty(stuObj.get("COLLEGE_NAME"))){
	        	    	remark +="所属学院不能为空 .";
	        	    	errcnt++;
	        	    }
	        	    if (Validate.isEmpty(stuObj.get("SPECIALTY_NAME"))){
	        	    	remark +="所属专业不能为空 .";
	        	    	errcnt++;
	        	    }
	        	    collegeCode = SysDataUtils.getCollegeCode(this.getSession(),String.valueOf(stuObj.get("COLLEGE_NAME")).trim());
	        	    if (Validate.isEmpty(collegeCode)){
	        	    	remark +="["+stuObj.get("COLLEGE_NAME")+"]学院名称在专业代码库不存在 .";
	        	    	errcnt++;
	        	    }
	        	    specialtyCode = getSpecialtyCode(String.valueOf(stuObj.get("SPECIALTY_NAME")).trim());
	        	    	//	SysDataUtils.getSpecialtyCode(this.getSession(),String.valueOf(stuObj.get("SPECIALTY_NAME")).trim());
	        	    if (Validate.isEmpty(specialtyCode)){
	        	    	remark +="["+stuObj.get("SPECIALTY_NAME")+"]专业名称在专业代码库不存在 .";
	        	    	errcnt++;
	        	    	failcnt++;
	        	    }
	        	    if (remark != null && !remark.isEmpty() ){
	        	    	stuObj.put("REMARK", remark);
			       	}
	        	    if(errcnt==0) {
	        	    	try{
	        	    		BaseSysClass baseSysClass = this.findSingle(BaseSysClass.class, DBCondition.form("classCode", String.valueOf(stuObj.get("CLASS_NO"))));
	        				if(baseSysClass==null) {
	        					baseSysClass = new BaseSysClass();
	        					baseSysClass.setClassCode(String.valueOf(stuObj.get("CLASS_NO")));
	        				}
	        	    		//填充值
	        				baseSysClass.setClassName(String.valueOf(stuObj.get("CLASS_NAME")));
	        				baseSysClass.setSpecialtyCode(specialtyCode);
	        				baseSysClass.setDataStatus("1");
	        				baseSysClass.setCollegeCode(collegeCode);
	        				baseSysClass.setGradeCode(yearNum);
	        	    		this.merge(baseSysClass);
	        	    		
	        	    		stuObj.put("REMARK", "班级[" + stuObj.get("CLASS_NAME")+ "]的数据库提交成功." );
    	    	            succnt++;
        	    		}catch (Exception e) {
        	    			e.printStackTrace();
                   	        stuObj.put("REMARK",  "班级[" + stuObj.get("CLASS_NAME")+ "]的数据库提交失败." );
            	    	    failcnt++;
						}
	        	    }else {
	        	    	failcnt++;
	        	    }
	    		}
	    	    //生成处理结果文件
		      // 	String rootPath = SingletonService.getRealPath().replace("/",System.getProperty("file.separator")).replace("\\", System.getProperty("file.separator"));
		       	String rootPath = Configurations.instance().getString("easy.upload.file.path").replace("/",System.getProperty("file.separator"));
		       	String zdypath = DicConst.IMPLEAVE_PATH+File.separator;
		       	String excelpath = rootPath + zdypath;;
		       	File fDbfpath =new File(excelpath);    
		        //如果文件夹不存在则创建     
		       	if  (!fDbfpath.exists()  && !fDbfpath.isDirectory()){      
		           Logs.log().info( "[" + excelpath + "]不存在,创建" ); 
		           fDbfpath .mkdir();
		       	} 
		       	String wdbffilename = DicConst.IMPLEAVE_PATH+"_CLASS_" + DateSource.getTradedate( CFrame.DatePattern5 ) + excelname;
		       	String wdbffilepath = excelpath + wdbffilename;
		       	String cdbffilepath = zdypath + wdbffilename;
		       	File outputFile = new File(wdbffilepath);

		       	// 需要特殊转换的单元
		       	Map<String, Map> excelHeadConvertMap = new java.util.HashMap<String, Map>();
		       	excelColumns.add(new ExcelColumn(excelColumns.size(), "REMARK", "备注"));
		       	head.setRowCount(1); // 模板中头部所占行数
		       	head.setColumns(excelColumns);  // 列的定义
		       	head.setColumnsConvertMap(excelHeadConvertMap); // 列的转换
		       
		       	ExcelHelper.getInstanse().exportExcelFile(head, fileIS, outputFile, classList);
	       		//返回数据
	       		surestream.put("resultinfo", "班级导入处理结果,成功[" + succnt +"]条,失败["+ failcnt +"]条请下载处理结果文件" );
	       		surestream.put("retfilename", wdbffilename);//处理结果文件名
	       		surestream.put("retfilepath", cdbffilepath);//处理结果文件存放路径
	       }
		}catch ( RWarn e) {
	        TryThrowable.agent( e, this );
	    }catch ( RException e) {
	        TryThrowable.agent( e, this );
	    }catch (Exception e) {
	        e.printStackTrace();
	    	TryThrowable.agent( e, this );
	    }
		//@output
		surestream.putReason(RWarn._000000, "导入操作成功");
		//@output
		return true;
	}
	
	private String getSpecialtyCode(String specialtyName) {
		String specialtyCode = this.queryForString(" SELECT SPECIALTY_CODE FROM  BASE_SYS_SPECIALTY WHERE SPECIALTY_NAME  =? ", specialtyName);
		if(Validate.isEmpty(specialtyCode)) {
			BaseSysSpecialty baseSysSpecialty = new BaseSysSpecialty();
			baseSysSpecialty.setDataStatus("1");
			//baseSysSpecialty.setGbSpecialtyCode(gbSpecialtyCode);
			baseSysSpecialty.setSpecialtyCode(DcKeyGenerate.getStringKeyCode("specialtyCode"));
			baseSysSpecialty.setSpecialtyName(specialtyName);
			//baseSysSpecialty.setXbSpecialtyCode(xbSpecialtyCode);
			this.save(baseSysSpecialty);
			specialtyCode=baseSysSpecialty.getSpecialtyCode();
		}
		return specialtyCode;
	}
